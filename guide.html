Архитектурные решения

✅ Neon вместо Docker Postgres
PostgreSQL живёт в облаке Neon — убираем один контейнер из Docker Compose. pgvector включается одной кнопкой в консоли. Branching — каждый разработчик получает свою ветку БД. В проде Neon масштабируется автоматически.
🔐 Backend закрыт от внешнего мира
Python FastAPI не имеет публичных портов. Браузер никогда не обращается к нему напрямую. Все запросы идут через Next.js Route Handlers → внутренняя Docker сеть → http://backend:8000.
📦 Два менеджера миграций
Prisma управляет бизнес-схемой (users, sessions, messages). Alembic управляет vector_chunks — Prisma не умеет работать с pgvector типом данных напрямую.

import asyncio
from rag_anything import RAGAnything
from app.core.embedder import embed_chunks
from app.storage.minio import download_to_tmp
from app.db.models import IndexStatus

_rag = RAGAnything()   # singleton, инициализируется один раз

async def ingest_document(file_id: str, storage_path: str):
    await set_status(file_id, IndexStatus.PROCESSING)
    try:
        # 1. Скачать оригинал из MinIO
        tmp_path = await download_to_tmp(storage_path)

        # 2. RAG-Anything: автоопределение формата, парсинг
        #    Поддерживает: PDF, DOCX, PPTX, XLSX, MD, TXT, HTML, изображения
        raw_chunks = await asyncio.to_thread(
            _rag.parse_document, str(tmp_path)
        )
        # raw_chunks: [{text, metadata: {page, type, bbox}}, ...]

        # 3. Батчевые embeddings (по 100 чанков, text-embedding-3-small)
        embedded = await embed_chunks(raw_chunks)

        # 4. Сохранить векторы в Neon pgvector
        await bulk_insert_chunks(file_id, embedded)
        await set_status(file_id, IndexStatus.READY, len(embedded))

    except Exception as e:
        await set_status(file_id, IndexStatus.ERROR)
        raise
backend/app/core/retriever.py
async def retrieve_chunks(query: str, top_k: int = 5) -> list[dict]:
    embedding = await embed_single(query)
    embed_str = "[" + ",".join(map(str, embedding)) + "]"

    async with async_session() as db:
        rows = await db.execute(text("""
            SELECT id, content, metadata,
                   1 - (embedding <=> :emb::vector) AS similarity
            FROM vector_chunks
            WHERE 1 - (embedding <=> :emb::vector) > 0.7
            ORDER BY embedding <=> :emb::vector
            LIMIT :k
        """), {"emb": embed_str, "k": top_k})
    return [dict(r) for r in rows.mappings()]

import json
from openai import AsyncOpenAI

client = AsyncOpenAI()

async def stream_response(messages, chunks, session_id):
    context = "\n\n---\n\n".join(
        f"[{c['metadata'].get('filename')} · стр.{c['metadata'].get('page')}]\n{c['content']}"
        for c in chunks
    )

    # Сначала отправляем источники
    sources = [{"id": c["id"], "filename": c["metadata"].get("filename"),
                "page": c["metadata"].get("page"), "sim": round(c["similarity"], 3)} for c in chunks]
    yield f"data: {json.dumps({'type': 'sources', 'data': sources})}\n\n"

    # Стриминг токенов gpt-4o
    full = ""
    stream = await client.chat.completions.create(
        model="gpt-4o", stream=True, temperature=0.1,
        messages=[{"role": "system", "content": HR_SYSTEM_PROMPT + context},
                  *messages],
    )
    async for chunk in stream:
        delta = chunk.choices[0].delta.content or ""
        full += delta
        yield f"data: {json.dumps({'type': 'token', 'data': delta})}\n\n"

    # done — Next.js сохранит в БД
    yield f"data: {json.dumps({'type': 'done', 'data': full, 'session_id': session_id})}\n\n"
backend/app/core/hallucination.py — LLM Judge
JUDGE_PROMPT = """Ты — система проверки качества RAG ответов.

ОТВЕТ МОДЕЛИ:
{answer}

ИСТОЧНИКИ (retrieved chunks):
{chunks}

Оцени: содержит ли ответ информацию, которой НЕТ в источниках?
Верни JSON: {{"score": 0.0-1.0, "reason": "..."}}
0.0 = нет галлюцинаций, 1.0 = полностью выдуманный ответ"""

async def check_hallucination(answer: str, chunks: list[dict]) -> float:
    chunks_text = "\n".join(c["content"] for c in chunks)
    res = await client.chat.completions.create(
        model="gpt-4o-mini",   # дешевле для judge
        messages=[{"role": "user",
                   "content": JUDGE_PROMPT.format(answer=answer, chunks=chunks_text)}],
        response_format={"type": "json_object"},
    )
    data = json.loads(res.choices[0].message.content)
    return data["score"]

import { auth } from "@/lib/auth"
import { NextResponse } from "next/server"

export default auth(function middleware(req) {
  const { nextUrl, auth: session } = req

  // Редирект неавторизованных
  if (!session && nextUrl.pathname.startsWith("/chat"))
    return NextResponse.redirect(new URL("/login", req.url))

  // Только ADMIN в /admin
  if (nextUrl.pathname.startsWith("/admin") && session?.user.role !== "ADMIN")
    return NextResponse.redirect(new URL("/chat", req.url))

  // Проверка TTL сессии (async в middleware через Edge Runtime)
  return NextResponse.next()
})

export const config = { matcher: ["/chat/:path*", "/admin/:path*"] }
frontend/app/api/chat/route.ts — прокси + стриминг
import { getServerSession } from "next-auth"
import { extendSessionIfNeeded } from "@/lib/session"
import { saveMessage } from "@/lib/db/messages"

export async function POST(req: Request) {
  const session = await getServerSession()
  if (!session) return new Response("Unauthorized", { status: 401 })

  // Продлить если < 1 часа
  await extendSessionIfNeeded(session.sessionId)

  const body = await req.json()
  const pythonRes = await fetch(
    `${process.env.PYTHON_BACKEND_URL}/api/v1/chat`,
    { method: "POST", headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ messages: body.messages, session_id: session.sessionId }) }
  )

  const stream = new ReadableStream({
    async start(controller) {
      const reader = pythonRes.body!.getReader()
      let buffer = ""
      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        buffer += new TextDecoder().decode(value)
        // Перехватить done-событие для сохранения в Neon
        for (const line of buffer.split("\n")) {
          if (!line.startsWith("data: ")) continue
          const evt = JSON.parse(line.slice(6))
          if (evt.type === "done")
            await saveMessage(session.sessionId, evt.data, evt.sources)
        }
        controller.enqueue(value)
      }
      controller.close()
    }
  })

  return new Response(stream, {
    headers: { "Content-Type": "text/event-stream", "Cache-Control": "no-cache" }
  })
}
frontend/lib/session.ts — TTL + toggle
const SIX_H = 6 * 3600 * 1000
const ONE_H = 1 * 3600 * 1000
const TWO_H = 2 * 3600 * 1000

export async function extendSessionIfNeeded(sessionId: string) {
  const s = await prisma.session.findUnique({ where: { id: sessionId } })
  if (!s?.persistent) return             // ephemeral — не продлеваем
  if (s.expiresAt.getTime() - Date.now() < ONE_H)
    await prisma.session.update({
      where: { id: sessionId },
      data: { expiresAt: new Date(Date.now() + TWO_H) }
    })
}

// Ephemeral режим: закрыл вкладку — сессия умерла
export function watchEphemeralSession() {
  document.addEventListener("visibilitychange", () => {
    if (document.visibilityState === "hidden")
      fetch("/api/session/terminate", { method: "POST", keepalive: true })
  })
}

Реализация всех фич

Детальное описание каждой из 7 фич с указанием где что живёт.

1
Admin + User роли
NextAuth v5 + Credentials provider. Поле role: ADMIN | USER в таблице User.
Middleware защищает /admin/* (только ADMIN) и /chat/* (активная сессия).
Admin создаётся через seed скрипт или invite-code механизм.
JWT strategy bcrypt role middleware
2
CRUD файлов базы знаний
Admin загружает файл → Next.js проксирует на POST /api/v1/ingest → FastAPI сохраняет в MinIO → background task запускает RAG-Anything pipeline → статус в Redis → клиент поллит /ingest/{id}/status.

Форматы: PDF DOCX PPTX XLSX TXT MD HTML
Delete: удаляет оригинал из MinIO + все vector_chunks из Neon.
3
Сессия 6 часов + автопродление
При логине создаётся запись в таблице Session с expiresAt = now + 6h.
На каждый запрос к /api/chat: если осталось < 1 часа — продлеваем на +2 часа.
История чата привязана к сессии — при возврате в течение 6ч пользователь видит весь контекст.
Middleware проверяет expiresAt и редиректит если сессия истекла.
4
Toggle: Persistent / Ephemeral сессия
Компонент SessionToggle.tsx — shadcn <Switch/> в шапке чата.
ON (persistent): сессия живёт 6ч, продлевается автоматически.
OFF (ephemeral): клиент слушает visibilitychange — при скрытии вкладки вызывает POST /api/session/terminate с keepalive: true.
Кнопка "Завершить сессию" — устанавливает terminatedAt, очищает cookie, редирект на /login.
5
Predefined Prompt Cards
Таблица PromptTemplate в Neon. Admin редактирует в /admin/prompts через CRUD интерфейс.
UI для пользователя: горизонтальный скролл карточек над полем ввода. Клик по карточке — вставляет текст в textarea.
Поля: title, content (шаблон с плейсхолдерами), category, order, isActive.
6
Feedback + Аналитика
После каждого ответа — кнопки 👍 / 👎 + опциональный текстовый комментарий (shadcn Popover).
Таблица Feedback в Neon, привязана к Message через @unique.
Admin Dashboard: общий % позитивных, recharts LineChart по дням, таблица негативных с контекстом вопроса, экспорт CSV.
7
Health Check + Оценка галлюцинаций
Страница /admin/health. Поллинг раз в 30 секунд к GET /api/v1/health/detailed.
Метрики FastAPI: ping OpenAI API (latency ms), кол-во чанков в pgvector, Redis ping.
Hallucination Judge: gpt-4o-mini оценивает каждый ответ асинхронно после стриминга. Score 0–1 сохраняется в поле hallScore таблицы Message. На дашборде — средний score за 24ч.

База данных — Neon

Два ORM: Prisma (бизнес-данные) и Alembic (векторные данные). Обе миграции применяются через make migrate.

Neon Setup — 3 шага
1. Зайти на neon.tech → New Project → выбрать регион ближайший к серверу
2. В консоли: Project Settings → Extensions → Enable pgvector
3. Connection Details → скопировать два URL: обычный (для Prisma) и asyncpg (для Python)
Neon Branching для команды
Каждый разработчик создаёт свою ветку БД: neon branches create --name dev-vasya
Получает отдельный DATABASE_URL. Изменения схемы не ломают коллег.
В CI/CD — автоматически создаётся ветка под каждый PR и удаляется после merge.
